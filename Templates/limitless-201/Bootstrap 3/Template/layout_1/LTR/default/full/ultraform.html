                    if (row.ObjJobPosition != null)
                        return '' + row.ObjJobPosition.Name + '';
                    return '';

                filter: {
                    type: 'option',
                    ajax: {
                        url: '/category/jobpositionlist',

                    ajax: {
                        url: '/employee/suggestions',

                                limit: 10
                            };
                        },

                                        text: this.FullName
                                    });
                                });
                            return {
                                results: data
                            };
                        }
